# Apply disk optimizations
for disk in /dev/sd[a-z]; do
    if [ -b "$disk" ]; then
        # Detect disk type
        if [[ $(cat /sys/block/$(basename $disk)/queue/rotational) -eq 0 ]]; then
            echo "none" > /sys/block/$(basename $disk)/queue/scheduler
            echo "128" > /sys/block/$(basename $disk)/queue/read_ahead_kb
        else
            echo "deadline" > /sys/block/$(basename $disk)/queue/scheduler
            echo "4096" > /sys/block/$(basename $disk)/queue/read_ahead_kb
        fi
    fi
done

        temp=$(smartctl -A $disk | grep "Temperature_Celsius" | awk '{print $10}')
        reallocated=$(smartctl -A $disk | grep "Reallocated_Sector_Ct" | awk '{print $10}')
        
        if [ "$health" != "PASSED" ]; then
            echo "CRITICAL: Disk $disk health check failed" | mail -s "NAS Disk Failure Alert" sysadmin@company.com
        fi
        
        if [ "$temp" -gt 50 ]; then
            echo "WARNING: Disk $disk temperature is $temp°C" | mail -s "NAS Disk Temperature Alert" sysadmin@company.com
        fi
        
        if [ "$reallocated" -gt 10 ]; then
            echo "WARNING: Disk $disk has $reallocated reallocated sectors" | mail -s "NAS Disk Reallocation Alert" sysadmin@company.com
        fi
    fi
done

4.5.2 Capacity Planning

# Capacity planning and forecasting
cat > /usr/local/bin/capacity-forecast << 'EOF'
#!/bin/bash
# Capacity planning tool

# Current usage
current_usage=$(df -h /mnt/nas | tail -1 | awk '{print $5}' | sed 's/%//')
total_space=$(df -h /mnt/nas | tail -1 | awk '{print $2}')
used_space=$(df -h /mnt/nas | tail -1 | awk '{print $3}')

# Growth rate analysis (30-day window)
growth_data="/var/log/mynas/capacity_growth.csv"
if [ -f "$growth_data" ]; then
    # Calculate daily growth rate
    growth_rate=$(tail -30 $growth_data | awk -F, '{sum+=$2} END {print sum/NR}')
else
    growth_rate=1  # Default 1% daily growth
fi

# Projections
echo "=== Capacity Forecast Report ==="
echo "Report Date: $(date)"
echo ""
echo "Current Status:"
echo "- Total Space: $total_space"
echo "- Used Space: $used_space ($current_usage%)"
echo "- Available: $(df -h /mnt/nas | tail -1 | awk '{print $4}')"
echo ""
echo "Growth Analysis:"
echo "- Daily Growth Rate: ${growth_rate}%"
echo ""
echo "Projections:"

for days in 30 60 90 180 365; do
    projected_usage=$(echo "scale=2; $current_usage * (1 + ($growth_rate/100 * $days))" | bc)
    if (( $(echo "$projected_usage > 100" | bc -l) )); then
        projected_usage=100
        exceed_date=$(date -d "+$((days-1)) days" "+%Y-%m-%d")
    fi
    
    if [ "$projected_usage" = "100" ]; then
        echo "- $days days: 100% (Exceeds capacity around $exceed_date)"
    else
        echo "- $days days: ${projected_usage}%"
    fi
done

echo ""
echo "Recommendations:"
if (( $(echo "$current_usage > 80" | bc -l) )); then
    echo "⚠️  CRITICAL: Immediate expansion required"
elif (( $(echo "$current_usage > 70" | bc -l) )); then
    echo "⚠️  WARNING: Plan for expansion within 30 days"
elif (( $(echo "$current_usage > 50" | bc -l) )); then
    echo "ℹ️  INFO: Monitor growth, consider expansion in 60-90 days"
else
    echo "✅ OK: Capacity sufficient for foreseeable future"
fi

# Log current usage
echo "$(date +%Y-%m-%d),$current_usage" >> $growth_data
EOF
